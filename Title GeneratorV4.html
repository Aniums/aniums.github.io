// --- UPDATED AUTO UPDATE CHECKER --- async function checkUpdates() { const msg = document.getElementById('update-msg'); let found = false; // We'll check in batches to avoid overwhelming the browser for (let i = CURRENT_VERSION + 1; i <= CURRENT_VERSION + 30; i++) { const targetUrl = `${BASE_URL}${i}.html`; try { // Using 'mode: no-cors' can sometimes bypass strict blocks, // but 'HEAD' is cleaner if the server allows it. const res = await fetch(targetUrl, { method: 'HEAD', mode: 'cors', cache: 'no-cache' }); if (res.ok) { LATEST_FOUND = targetUrl; found = true; } } catch (e) { // If fetch fails entirely, the file likely doesn't exist } } if (found) { msg.textContent = "There is a newer version for this generator"; document.getElementById('updateActions').style.display = "flex"; } else { msg.textContent = "You have the latest link to the generator."; } } checkUpdates();
